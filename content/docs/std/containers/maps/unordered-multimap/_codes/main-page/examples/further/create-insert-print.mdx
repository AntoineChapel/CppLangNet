#include <unordered_multimap>
#include <iostream>
#include <string>

int main() {
  std::unordered_multimap<std::string, std::string> player_equipment{
    {"player2", "Sword"},
    {"player2", "Health potion"}
  };

  player_equipment.emplace("player3", "Shield");

  for(const auto& [key, value] : player_equipment)
    std::cout << key << " has got " << value << std::endl;

  return 0;
}
